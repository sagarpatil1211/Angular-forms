ts file

new FormControl(value,validator)
Validators.compose is used to add more than one validator i.e. it is array of validators

email : new FormControl("",Validators.compose([Validators.required, Validators.email]))

                ngOnInit(): void 
{
    this.loginData = new FormGroup(
      {
// email : new FormControl("",Validators.compose([Validators.required, Validators.email])), // required validator

email : new FormControl("",Validators.compose([Validators.required,Validators.email, Validators.pattern("[a-zA-Z0-9.-_]{1,}@[a-zA-Z.-]{2,}
                                                                        [.]{1}[a-zA-Z]{3,}")])),  // using pattern

password : new FormControl ("",Validators.compose([Validators.required])) 
       
      })
}
